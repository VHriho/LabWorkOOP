import java.util.Arrays;

public class Matrix implements IMatrix {

    private final float[][] elements;

    //РєРѕРЅСЃС‚СЂСѓРєС‚РѕСЂ, С‰Рѕ СЃС‚РІРѕСЂСЋС” РїСѓСЃС‚Сѓ РјР°С‚СЂРёС†СЋ
    public Matrix() {
        elements = new float[0][0];
    }

    //РєРѕРЅСЃС‚СЂСѓРєС‚РѕСЂ, С‰Рѕ СЃС‚РІРѕСЂСЋС” РјР°С‚СЂРёС†СЋ Р·Р°РґР°РЅРѕРіРѕ СЂРѕР·РјС–СЂСѓ
    public Matrix(int m, int n) {
        if (m < 0 || n < 0)
            throw new NegativeArraySizeException("Row or column value is negative");
        else
            elements = new float[m][n];
    }

    //РєРѕРЅСЃС‚СЂСѓРєС‚РѕСЂ, С‰Рѕ СЃС‚РІРѕСЂСЋС” РєРѕРїС–СЋ С–РЅС€РѕС— РјР°С‚СЂРёС†С–
    public Matrix(IMatrix copiedMatrix) {
        elements = new float[copiedMatrix.getRows()][copiedMatrix.getColumns()];
        for (int i = 0; i < copiedMatrix.getRows(); i++)
            elements[i] = Arrays.copyOf(copiedMatrix.getMatrix()[i], copiedMatrix.getMatrix()[i].length);
    }

    //РїРµСЂРµРІРёР·РЅР°С‡РµРЅРёР№ РјРµС‚РѕРґ equals
    @Override
    public boolean equals(Object other) {
        if(this == other)
            return true;
        if(getClass() != other.getClass())
            return false;
        Matrix newMatrix = (Matrix) other;
        return (elements.length == newMatrix.elements.length && elements[0].length == newMatrix.elements[0].length && Arrays.deepEquals(elements, newMatrix.elements));

    }

    //РїРѕРІРµСЂС‚Р°С” РєС–Р»СЊРєС–СЃС‚СЊ СЂСЏРґРєС–РІ
    @Override
    public int getRows(){
        return elements.length;
    }

    //РїРѕРІРµСЂС‚Р°С” РєС–Р»СЊРєС–СЃС‚СЊ СЃС‚РѕРІРїС†С–РІ
    @Override
    public int getColumns(){
        return elements[0].length;
    }

    //РџРѕРІРµСЂС‚Р°С” РјР°С‚СЂРёС†СЋ
    @Override
    public float[][] getMatrix() {
        return elements;
    }

}
